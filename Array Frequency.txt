#include <stdio.h>
#include <string.h>
               //Digit Frequency
int main() {
    char str[1000];
    int freq[10] = {0};   // array to store frequency of digits 0-9

    // read input string
    scanf("%s", str);

    // iterate through string
    for (int i = 0; i < strlen(str); i++) {
        if (str[i] >= '0' && str[i] <= '9') {
            freq[str[i] - '0']++;   // increment corresponding digit count
        }
    }

    // print frequencies
    for (int i = 0; i < 10; i++) {
        printf("%d", freq[i]);
        if (i < 9) printf(" ");   // space between numbers
    }
    printf("\n");

    return 0;
}
