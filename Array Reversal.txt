#include <stdio.h>
          //Array Reversal
int main() {
    int n;
    scanf("%d", &n);   // read size of array

    int arr[n];        // declare array of size n

    // read array elements
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // print array in reverse order
    for (int i = n - 1; i >= 0; i--) {
        printf("%d", arr[i]);
        if (i > 0) printf(" ");  // space between numbers
    }
    printf("\n");

    return 0;
}
