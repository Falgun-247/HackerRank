#include <stdio.h>
#include <stdlib.h>
#include <string.h>
                        //Post Transition
typedef struct package {
    char* id;
    int weight;
} package;

typedef struct post_office {
    int packages_count;
    package* packages;
    int min_weight;
    int max_weight;
} post_office;

typedef struct town {
    char* name;
    int offices_count;
    post_office* offices;
} town;

void print_all_packages(town t) {
    printf("%s:\n", t.name);
    for (int i = 0; i < t.offices_count; i++) {
        printf("\t%d:\n", i);
        for (int j = 0; j < t.offices[i].packages_count; j++) {
            printf("\t\t%s\n", t.offices[i].packages[j].id);
        }
    }
}

void send_all_acceptable_packages(town* t, int source, town* t2, int target) {
    post_office* from = &t->offices[source];
    post_office* to = &t2->offices[target];

    // Arrays to store accepted and rejected packages
    package* accepted = malloc(from->packages_count * sizeof(package));
    package* rejected = malloc(from->packages_count * sizeof(package));

    int acc = 0, rej = 0;

    // Separate packages
    for (int i = 0; i < from->packages_count; i++) {
        package p = from->packages[i];
        if (p.weight >= to->min_weight && p.weight <= to->max_weight) {
            accepted[acc++] = p;
        } else {
            rejected[rej++] = p;
        }
    }

    // Append accepted to destination
    to->packages = realloc(to->packages, (to->packages_count + acc) * sizeof(package));
    for (int i = 0; i < acc; i++) {
        to->packages[to->packages_count++] = accepted[i];
    }

    // Replace source with rejected
    free(from->packages);
    from->packages = rejected;
    from->packages_count = rej;

    free(accepted);
}

town* find_town(town* towns, int towns_count, char* name) {
    for (int i = 0; i < towns_count; i++) {
        if (strcmp(towns[i].name, name) == 0) {
            return &towns[i];
        }
    }
    return NULL;
}

town* town_with_most_packages(town* towns, int towns_count) {
    int max_packages = -1;
    town* result = NULL;

    for (int i = 0; i < towns_count; i++) {
        int total = 0;
        for (int j = 0; j < towns[i].offices_count; j++) {
            total += towns[i].offices[j].packages_count;
        }
        if (total > max_packages) {
            max_packages = total;
            result = &towns[i];
        }
    }
    return result;
}

int main() {
    int towns_count;
    scanf("%d", &towns_count);

    town* towns = malloc(towns_count * sizeof(town));

    for (int i = 0; i < towns_count; i++) {
        towns[i].name = malloc(100 * sizeof(char));
        scanf("%s", towns[i].name);

        scanf("%d", &towns[i].offices_count);
        towns[i].offices = malloc(towns[i].offices_count * sizeof(post_office));

        for (int j = 0; j < towns[i].offices_count; j++) {
            post_office* office = &towns[i].offices[j];
            scanf("%d %d %d", &office->packages_count, &office->min_weight, &office->max_weight);

            office->packages = malloc(office->packages_count * sizeof(package));

            for (int k = 0; k < office->packages_count; k++) {
                office->packages[k].id = malloc(100 * sizeof(char));
                scanf("%s", office->packages[k].id);
                scanf("%d", &office->packages[k].weight);
            }
        }
    }

    int queries;
    scanf("%d", &queries);

    while (queries--) {
        int type;
        scanf("%d", &type);

        if (type == 1) {
            char name[100];
            scanf("%s", name);
            town* t = find_town(towns, towns_count, name);
            print_all_packages(*t);
        }
        else if (type == 2) {
            char town1[100], town2[100];
            int o1, o2;
            scanf("%s %d %s %d", town1, &o1, town2, &o2);

            town* t1 = find_town(towns, towns_count, town1);
            town* t2 = find_town(towns, towns_count, town2);

            send_all_acceptable_packages(t1, o1, t2, o2);
        }
        else if (type == 3) {
            town* t = town_with_most_packages(towns, towns_count);
            printf("Town with the most number of packages is %s\n", t->name);
        }
    }

    return 0;
}
